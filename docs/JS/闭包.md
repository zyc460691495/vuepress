title: 函数闭包
description:
---

```js
for (var i = 0; i < 5; i++) {
    setTimeout(function () {
        console.log(i++);
    }, 4000)
}
console.log(i);
// 5
// 5
// 6
// 7
// 8
// 9
```

```js
for (var i = 0; i < 5; i++) {
    (function (x) {
        setTimeout(function () {
            console.log(x++);
        }, 4000)
    })(i);
}
console.log(i);
// 5
// 0
// 1
// 2
// 3
// 4
```
